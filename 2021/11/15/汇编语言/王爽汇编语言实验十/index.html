王爽《汇编语言》实验十

发表于 2021-11-15 更新于 2022-01-16 分类于 汇编语言 ， 8086实验 阅读次数： Valine：

前言

  实验十属于第十章的内容：CALL和RET指令的讲解，引入子程序的概念与设计。其实这两个指令和之前讲的jmp指令十分类似，只不过多了把IP值或者CS和IP的值压栈出栈的过程，详细内容可以看书学习。

  实验十的要求是设计三个子程序，具体要求见下面。

题目

子程序1：显示字符串

  函数功能十分明确，用来显示字符串。要求是提供灵活的接口，要能够让用户自定义内容、位置（行、列）还有颜色。

  

  作者也十分贴心的给出了应用实例（这里就不贴了）和一些注意点：

  • 子程序的入口参数是屏幕上的行号和列号，在子程序内部我们要将它们转化成显存中的地址，一定要搞清楚两者的对应关系。
  • 注意保护子程序中用到的相关的寄存器。
  • 这个子程序的处理过程和显存的结构密切相关，但是向外提供了与显存结构无关的接口。通过这个接口，我们封装了这个子程序，不必向user透露显存的结构，为编程提供了方便，要注意这种思想。

子程序2：解决除法溢出的问题

  我们首先来回顾一下div指令，当进行8位除法时，用al存储结果的商，ah存储结果的余数；当进行16位除法时，用ax存储结果的商，dx存储结果的余数。

  但是如果结果的商>ax或者al所能存储的最大值该怎么办？

  这个时候就会引发CPU的一个内部错误：除法溢出。我们要设计一个程序来处理这个错误。不过我们现在还不能处理错误，得等学到后面才能处理；在这里我们想办法在做除法之前，注意除数和被除数，让他们的结果不会发生溢出。

子程序3：数值显示

  这个程序意思很明确，就是把各种不同进制的数都转化成十进制数然后打印在屏幕上。这个过程需要两步：

  1. 将用二进制存储的数据转化为十进制形式的字符串。
  2. 显示十进制形式的字符串。

解答

子程序1：显示寄存器

  我们先贴代码：(注意中文注释是我在markdown文件里加上去的，汇编代码文件中并没有中文注释，如果给汇编代码文件加上中文注释会显示乱码)

assume cs:code,ds:data,ss:stack
data segment
        db 'YangFan is a good boy!',0
data ends

stack segment
        dd 32 dup(0)
stack ends

code segment
main:   mov dh,13
        mov dl,18
        mov cl,2
        ;定义参数，dh表示行数，dl表示列数，cl表示颜色:1(01H)是蓝色，2(02H)是绿色.4(04H)是红色
        mov ax,data
        mov ds,ax
        mov ax,stack
        mov ss,ax
        mov sp,20h
        mov si,0
        ;初始化一些用于寻址的寄存器
        call show_str

        mov ax,4c00h
        int 21h
        ;结束程序
show_str: push es
          push ax
          push si
          push di
          push bx
          ;在执行子程序前，注意保护子程序使用到的寄存器原来的值
          mov di,0
          mov ax,0B800h
          mov es,ax
          ;将彩色显示区的基地址送入es段寄存器

          sub dh,1
          mov al,dh
          mov ah,00h
          push cx
          mov cx,00a0h
          mul cl
          pop cx
          mov bx,ax
          ;计算第13行开头的偏移地址，放入bx

          add dl,dl
          sub dl,2
          mov dh,00h
          add bx,dx
          ;计算第18列的偏移地址，然后加到bx中
          mov ah,cl
          ;将颜色的编码移动到ah，准备和字符组合，然后转移到彩色显示区

str_loop: mov al,[si]

          push cx
          mov cl,al
          mov ch,00h
          jcxz str_ends
          ;这部分判断是不是0，如果是0，显示结束
          pop cx
          ;别忘了pop cx的值，虽然在这个程序不影响，但是我们要坚持原则，保护寄存器的值
          mov es:[bx+di],ax
          inc si
          add di,2
          ;由于si是用于字符的寻址，而一个ASCII编码的字母是一个字节长，所以si每次+1;而di是用于彩色显示区的寻址，彩色显示区一个显示单元是一个字，所以di每次+2
          jmp near ptr str_loop

str_ends: pop cx
          pop bx
          pop di
          pop si
          pop ax
          pop es

          ret
code ends
end main

子程序2：解决除法溢出的问题

assume cs:code,ss:stack

stack segment
        dw 8 dup (0)
stack ends

code segment
start:  mov ax,stack
        mov ss,ax
        mov sp,16
        mov ax,4240h
        mov dx,00fh
        mov cx,0ah
        call divdw
        mov ax,4c00h
        int 21h

divdw:  push ax
        mov ax,dx
        mov dx,0
        div cx

        mov bx,ax
        pop ax
        div cx
        mov cx,dx
        mov dx,bx
        ret
code ends
end start

子程序3：数值显示

  第二步我们已经完成了，现在只要完成第一步就行。下面是完整的程序：

assume cs:code,ds:data,ss:stack
data segment
        db 32 dup(0)
data ends

stack segment
        dd 32 dup(0)
stack ends

code segment
main:    mov ax,12666
        mov bx,data
        mov ds,bx
        mov bx,stack
        mov ss,bx
        mov sp,20H
        mov si,0
        call dtoc

        mov sp,30H
        mov dh,8
        mov dl,3
        mov cl,2
        call show_str

        mov ax,4c00h
        int 21h

dtoc:    push ax
        push bx
        push cx
        push dx
        push si 
        mov bx,0

d_loop:         mov cx,10
                mov dx,0000h
        div cx
        add dx,30H
        push dx
        inc bx
        mov cx,ax
        jcxz d_end
        jmp near ptr d_loop

d_end:    mov cx,bx
display: pop ds:[si]
        inc si
        loop display

        mov ax,0
        mov [si],ax

        pop si
        pop dx
        pop cx
        pop bx
        pop ax
        ret

show_str: push es
          push ax
          push si
          push di
          push bx

          mov di,0
          mov ax,0B800h
          mov es,ax
          ;put the address of display area to es

          sub dh,1
          mov al,dh
          mov ah,00h
          push cx
          mov cx,00a0h
          mul cl
          pop cx
          mov bx,ax
          ;compute the bias of the row

          add dl,dl
          sub dl,2
          mov dh,00h
          add bx,dx
          ;compute the bias of the col
          mov ah,cl

str_loop: mov al,[si]
          push cx
          mov cl,al
          mov ch,00h
          jcxz str_ends
          pop cx
          mov es:[bx+di],ax
          inc si
          add di,2
          jmp near ptr str_loop

str_ends: pop cx
          pop bx
          pop di
          pop si
          pop ax
          pop es

          ret

code ends
end main
