жӢ·иҙқжһ„йҖ еҮҪж•°дёҺзұ»е°ҒиЈ…й—®йўҳ

еҸ‘иЎЁдәҺ2022-03-16|жӣҙж–°дәҺ2022-03-26|PLCpp
|еӯ—ж•°жҖ»и®Ў:415|йҳ…иҜ»йҮҸ:

вҖғвҖғжңҖиҝ‘еңЁзңӢжӢ·иҙқжһ„йҖ еҮҪж•°ж—¶жғіеҲ°дёҖдёӘдёҺзұ»е°ҒиЈ…зҡ„е°Ҹй—®йўҳгҖӮеҒҮи®ҫжҲ‘зҺ°еңЁеҶҷдәҶдёӢйқўиҝҷж ·дёҖдёӘзұ»пјҡ

class Student
{
    public:
    //еҲқе§ӢеҢ–жһ„йҖ еҮҪж•°
    Student(string Name,int Id) :name(Name),id(Id) {}
    //жӢ·иҙқжһ„йҖ еҮҪж•°
    Student(const Student &Stu)
    {
        name = Stu.name;
        id = Stu.id;
    }
    //жҲҗе‘ҳеҮҪж•°зӨәдҫӢ
    string GetName() {return name;}
    private:
    string name;
    int id;
};

вҖғвҖғжіЁж„ҸжӢ·иҙқжһ„йҖ еҮҪж•°жҳҜдҫқйқ еҗҢзұ»зҡ„е…¶е®ғеҜ№иұЎStuжқҘеҲқе§ӢеҢ–зҡ„пјҢжҜ”еҰӮиҜҙдёӢйқўиҝҷж ·пјҢStu2йҖҡиҝҮжӢ·иҙқжһ„йҖ еҮҪж•°дҪҝз”ЁStu1зҡ„еҖјжқҘеҲқе§ӢеҢ–пјҡ

int main()
{
    Student Stu1("SailYang",10);
    Student Stu2(Stu1);
    cout << Stu2.GetName() << endl;
    return 0;
}

вҖғвҖғйӮЈд№Ҳе°ұжңүдёҖдёӘй—®йўҳпјҡStudentзұ»зҡ„жҲҗе‘ҳnameе’ҢidдёҚжҳҜprivateеҗ—пјҹдёәд»Җд№ҲиғҪз”ЁжӢ·иҙқжһ„йҖ еҮҪж•°и®ҝй—®иҝҷдәӣз§ҒжңүеҸҳйҮҸе‘ўпјҹ

вҖғвҖғдёҖиҲ¬жқҘиҜҙпјҢз§ҒжңүеҸҳйҮҸеҸӘжңүйҖҡиҝҮpublicеҮҪж•°жүҚиғҪи®ҝй—®еҲ°пјҢиҖҢеңЁзұ»еҶ…йғЁжҳҜеҸҜд»ҘиҮӘз”ұи®ҝй—®зҡ„гҖӮиҝҷз§ҚдҝқжҠӨе…¶е®һжҳҜй’ҲеҜ№е…¶е®ғзұ»зҡ„пјҢиҖҢдёҚжҳҜеҗҢзұ»зҡ„дёҚеҗҢеҜ№иұЎпјҢеҚіеҗҢзұ»зҡ„дёҚеҗҢеҜ№иұЎжҳҜеҸҜд»ҘзӣҙжҺҘи®ҝй—®еҗ„иҮӘзҡ„з§ҒжңүеҸҳйҮҸзҡ„гҖӮ

вҖғвҖғдёҫдёӘдҫӢеӯҗ,жҲ‘д»¬еңЁStudentзұ»дёӯеҠ дёӘжҲҗе‘ҳеҮҪж•°Func():

void Func(const Student &Stu)
{
    cout << Stu.name() << endl;
}

вҖғвҖғз„¶еҗҺжҲ‘д»¬и°ғз”ЁдёҖдёӢпјҡ

вҖғвҖғеҸҜд»ҘзңӢеҲ°Stu2жҳҜиғҪи®ҝй—®еҗҢеұһдәҺдёҖдёӘзұ»зҡ„Stu1зҡ„з§ҒжңүжҲҗе‘ҳгҖӮжҲ‘д»¬еҒҮеҰӮжҠҠFuncиҝҷдёӘеҮҪж•°е®ҡд№үеңЁStudentзұ»еӨ–йқўпјҢеҶҚеҺ»и°ғз”ЁдёҖдёӢпјҢз»“жһңеҰӮдҪ•е‘ўпјҹ

вҖғвҖғйҖҡиҝҮеҜ№жҜ”пјҢе°ұиғҪеҫ—еҮәжҲ‘д»¬д№ӢеүҚзҡ„з»“и®әгҖӮе…¶е®һжғіжғізңӢпјҢж—ўз„¶е’ұд»¬йғҪеұһдәҺдёҖдёӘзұ»пјҢйӮЈиӮҜе®ҡйғҪзҹҘж №зҹҘеә•пјҢжңүеҝ…иҰҒи—ҸзқҖжҺ–зқҖеҗ—рҹҳ…
